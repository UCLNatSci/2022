2.4.1. Homework¶

import numpy as np
import matplotlib.pyplot as plt

n = 300
k1 = .2
k2 = .4
k3 = .1
k4 = .3

for k1 in np.linspace(0, 0.3, 20):

    X = np.zeros(n)
    X[0] = 0
    Y = np.zeros(n)
    Y[0] = 0

    for i in range(n - 1):
        X[i+1] = X[i] + k1 - k2*X[i] + k3*X[i]**2*Y[i]
        Y[i+1] = Y[i] + k4*X[i] - k3*X[i]**2*Y[i]

    plt.figure(figsize=(5,5))
    plt.plot(X, Y)

    plt.figure(figsize=(5,5))
    plt.plot(X)
    plt.plot(Y)
